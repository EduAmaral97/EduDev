SELECT
CASE
    WHEN B.BA3_IDBENN <> '' THEN B.BA3_IDBENN
    WHEN B.BA3_IDBENN = '' AND A.BQC_SUBCON = '000000001' THEN B.BA3_MATEMP
    ELSE A.BQC_SUBCON
END AS IDCONTRATO,
CASE
    WHEN A.BQC_SUBCON = '000000001' THEN B.BA3_EQUIPE
    ELSE A.BQC_EQUIPE
END AS EQUIPE_CAPA,
C.BA1_IDBENN AS IDBENNER,
E.BT5_NOME   AS PERFIL,
C.BA1_NOMUSR AS BENEFICIARIO,
C.BA1_EQUIPE AS EQUIPE_BENEF,
EQ.BXL_DESEQU AS DESC_EQUIPE,
C.BA1_DATBLO,
C.BA1_MOTBLO
FROM BQC010 A
LEFT JOIN BA3010 B ON B.BA3_CODINT = A.BQC_CODINT AND B.BA3_CODEMP = A.BQC_CODEMP AND B.BA3_CONEMP = A.BQC_NUMCON AND B.BA3_SUBCON = A.BQC_SUBCON AND B.D_E_L_E_T_ = ''   
LEFT JOIN BA1010 C ON C.BA1_FILIAL = B.BA3_FILIAL AND C.BA1_CODINT = B.BA3_CODINT AND C.BA1_CODEMP = B.BA3_CODEMP AND C.BA1_CONEMP = B.BA3_CONEMP AND C.BA1_SUBCON = B.BA3_SUBCON AND C.BA1_MATEMP = B.BA3_MATEMP AND C.D_E_L_E_T_ = ''   
LEFT JOIN BXL010 EQ ON EQ.D_E_L_E_T_ = '' AND EQ.BXL_CODEQU = C.BA1_EQUIPE 
LEFT JOIN BT5010 E ON E.D_E_L_E_T_ = '' AND E.BT5_FILIAL = A.BQC_FILIAL AND E.BT5_CODINT = A.BQC_CODINT AND E.BT5_CODIGO = A.BQC_CODEMP AND E.BT5_NUMCON = A.BQC_NUMCON AND E.BT5_VERSAO = A.BQC_VERCON
WHERE 1=1
AND C.BA1_EQUIPE <> ''
AND C.BA1_DATBLO = ''
AND C.BA1_MOTBLO = ''

2 - LIMPAR CAPA FAMILIA PF
3 - LIMPAR BA1 DOS BENEFICIARIO QUE POSSUEM IDBENNER
